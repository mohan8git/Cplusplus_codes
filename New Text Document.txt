#include <iostream>
using namespace std;

int main() {
	string a;
	getline(cin,a);
	for(int i=0;i<a.size();i++)
	  {
	      if(a[i]>=65 && a[i]<=90)
	      cout<<char(a[i]+32);
	      
	      else if(a[i]>=97 && a[i]<=122)
	      cout<<char(a[i]-32);
	      else if(isblank(a[i]))
	      cout<<" ";
	  }
	return 0;
}

/* strncmp example */
#include <stdio.h>
#include <string.h>

int main ()
{
  char str[][5] = { "R2D2" , "C3PO" , "R2A6" };
  int n;
  puts ("Looking for R2 astromech droids...");
  for (n=0 ; n<3 ; n++)
    if (strncmp (str[n],"R2xx",2) == 0)
    {
      printf ("found %s\n",str[n]);
    }
  return 0;
}
